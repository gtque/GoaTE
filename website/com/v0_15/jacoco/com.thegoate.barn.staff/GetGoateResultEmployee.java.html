package com.thegoate.barn.staff;

import com.thegoate.staff.Employee;
import com.thegoate.staff.GoateJob;

/**
 * Gets a specific result number.
 * Useful when used as part of expectations for an integration or multistep test case.
 * Created by Eric Angeli on 5/22/2017.
 */
@GoateJob(jobs = {"get result"})
public class GetGoateResultEmployee extends Employee {

    Object result = null;

    @Override
    public Employee init() {
        result = definition.get("_goate_result");
        return this;
    }

    @Override
    protected Object doWork() {
        return result;
    }

    @Override
    public String[] detailedScrub(){
        String[] scrub = {"_goate_result"};
        return scrub;
    }
}
