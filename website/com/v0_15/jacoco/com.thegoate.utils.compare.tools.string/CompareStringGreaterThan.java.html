package com.thegoate.utils.compare.tools.string;

import com.thegoate.annotations.IsDefault;
import com.thegoate.utils.compare.CompareUtil;

import static com.thegoate.utils.type.GoateNullCheck.isNull;

/**
 * Checks if a string is lexographically greater than.
 * Created by Eric Angeli on 4/17/2018.
 */
@CompareUtil(operator = ">", type = String.class)
@IsDefault
public class CompareStringGreaterThan extends CompareStringTool {
    public CompareStringGreaterThan(Object actual) {
        super(actual);
    }

    @Override
    public boolean evaluate() {
        String check = ""+actual;
        String act = "" + expected;
        return isNull(actual)?(isNull(expected)?false:true):(isNull(expected)?false:check.compareTo(act)>0);
    }
}
