package com.thegoate.utils.get;

import com.thegoate.Goate;

/**
 * Loads the file specified in from into a string and returns it.
 * Created by Eric Angeli on 5/5/2017.
 */
@GetUtil(type = Goate.class)
public class GetGoate extends GetTool{

    public GetGoate(){
        super(new Goate());
    }

    public GetGoate(Object selector) {
        super(selector);
    }

    @Override
    public boolean isType(Object check) {
        return check instanceof Goate;
    }

    @Override
    public Object from(Object container) {
        Object result = new NotFound(""+selector);
        if(container instanceof Goate){
            if(((Goate)container).getStrict(""+selector)==null){
                result = new NotFound(""+selector);
            } else {
                result = ((Goate) container).get("" + selector);
            }
        }
        result = processNested(result);//process nested gets.
        return result;
    }
}
