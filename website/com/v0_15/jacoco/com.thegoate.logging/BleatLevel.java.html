        this.name = name;
        this.value = value;
    }

    public static BleatLevel level(Level level, String name){
        BleatLevel volume = new BleatLevel(level.getName(), level.intValue());
        return volume.tune(name);
    }

        return new BleatLevel(name, volume);
    }
